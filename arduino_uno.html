#include <Servo.h>
Servo myservo;
void setup() {
  Serial.begin(9600);
  // put your setup code here, to run once:
  pinMode(12,OUTPUT);
  pinMode(13,OUTPUT);
  //For fan:
  pinMode(9,OUTPUT);
  pinMode(3,OUTPUT);
  pinMode(2,OUTPUT);
  //For led:
  pinMode(8,INPUT);
  //For Motion Sensor:
  pinMode(5,OUTPUT);
  //For Servo Motor:
  pinMode(A0,INPUT);
  //For Ir Sensor:
myservo.attach(5);
}

void loop() {
  bool motion=digitalRead(8);    
  Serial.println(motion);
  int sensor=analogRead(A0);  
  Serial.println(sensor);
  // put your main code here, to run repeatedly:
  if (digitalRead(8)){
  myservo.write(180);
  delay(500);
  myservo.write(90);
  delay(500);
  }
  else{
    myservo.write(0);
    
  }
  if(sensor>100){
    digitalWrite(9,HIGH);
    digitalWrite(2,LOW);
    digitalWrite(3,LOW);  
  }
  else{
    if(sensor>10 && sensor<50){
         digitalWrite(9,LOW);
         digitalWrite(2,HIGH);
         digitalWrite(3,LOW);
    }
    else if(sensor>50 && sensor<100){
      digitalWrite(9,LOW);
      digitalWrite(2,LOW);
      digitalWrite(3,HIGH);
    }
    else{
      digitalWrite(9,LOW);
      digitalWrite(2,LOW);
      digitalWrite(3,LOW); 
    }
  }
   if(sensor>100){    
    digitalWrite(12,HIGH);
    digitalWrite(13,LOW);
    delay(3000);
    }
    else{
    digitalWrite(12,LOW);
    digitalWrite(13,LOW);
      }
}
